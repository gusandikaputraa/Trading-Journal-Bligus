Langkah Setup
1
Buka Google Sheets baru di sheets.new, beri nama "Trading Journal"
2
Di menu atas, klik Extensions → Apps Script
3
Hapus semua kode yang ada, lalu paste kode berikut:
// ============================================================ // TRADING JOURNAL v2 — Google Apps Script // Mendukung: Multi Account, Balance Tracking, Close Time, Durasi, Lot Size // ============================================================ const SHEET_TRADES = "Trades"; const SHEET_ACCOUNTS = "Accounts"; const SHEET_LOT_SIZES = "LotSizes"; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const action = data.action || "save_trade"; if (action === "save_trade") return saveTrade(ss, data); if (action === "save_account") return saveAccount(ss, data); if (action === "update_account_balance") return updateAccountBalance(ss, data); if (action === "update_trade") return updateTrade(ss, data); if (action === "save_lot_size") return saveLotSize(ss, data); return jsonResponse({status:"error", message:"Unknown action"}); } catch(err) { return jsonResponse({status:"error", message: err.toString()}); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Jika ada payload (write via GET untuk hindari CORS) if (e.parameter.payload) { const data = JSON.parse(e.parameter.payload); const action = data.action || ""; if (action === "save_trade") return saveTrade(ss, data); if (action === "save_account") return saveAccount(ss, data); if (action === "update_account_balance") return updateAccountBalance(ss, data); if (action === "update_trade") return updateTrade(ss, data); if (action === "save_lot_size") return saveLotSize(ss, data); return jsonResponse({status:"error", message:"Unknown action: " + action}); } const action = e.parameter.action || "ping"; if (action === "get_trades") return getTrades(ss, e.parameter); if (action === "get_accounts") return getAccounts(ss); if (action === "get_lot_sizes") return getLotSizes(ss); return jsonResponse({status:"ok", message:"Trading Journal API v3 aktif"}); } catch(err) { return jsonResponse({status:"error", message: err.toString()}); } } function saveLotSize(ss, data) { let sheet = ss.getSheetByName(SHEET_LOT_SIZES); if (!sheet) { sheet = ss.insertSheet(SHEET_LOT_SIZES); const headers = ["ID","Pair","Jarak Harga","Pip Size","Nominal per Lot","Currency","Catatan","Updated At"]; sheet.getRange(1,1,1,headers.length).setValues([headers]) .setBackground("#1a1a2e").setFontColor("#00d4aa").setFontWeight("bold"); sheet.setFrozenRows(1); } // Check if pair already exists -> update const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][1].toString().toUpperCase() === data.pair.toUpperCase()) { sheet.getRange(i+1, 1, 1, 8).setValues([[ data.id, data.pair, data.priceDistance, data.pipSize, data.nominal, data.currency, data.notes||"", new Date().toLocaleString("id-ID") ]]); return jsonResponse({status:"success", updated: data.pair}); } } sheet.appendRow([data.id, data.pair, data.priceDistance, data.pipSize, data.nominal, data.currency, data.notes||"", new Date().toLocaleString("id-ID")]); sheet.autoResizeColumns(1, 8); return jsonResponse({status:"success", pair: data.pair}); } function getLotSizes(ss) { const sheet = ss.getSheetByName(SHEET_LOT_SIZES); if (!sheet) return jsonResponse({status:"ok", lotSizes:[]}); const data = sheet.getDataRange().getValues(); const headers = data[0]; const lotSizes = data.slice(1).map(row => { const obj = {}; headers.forEach((h,i) => obj[h] = row[i]); return obj; }); return jsonResponse({status:"ok", lotSizes}); } function saveTrade(ss, data) { let sheet = ss.getSheetByName(SHEET_TRADES); if (!sheet) { sheet = ss.insertSheet(SHEET_TRADES); const headers = [ "ID","Tanggal","Jam Entry","Pair","Arah", "Entry Price","Stop Loss","Take Profit","Lot Size","Risk %", "Jenis Trading","Strategi","Ket.Strategi","Timeframe","Analisa", "Completed","Hasil","Close Price","P&L","Currency", "Tgl Close","Jam Close","Durasi","Emosi","Evaluasi", "Akun ID","Nama Akun","Broker","Saldo Setelah","Timestamp" ]; sheet.getRange(1,1,1,headers.length).setValues([headers]) .setBackground("#1a1a2e").setFontColor("#00d4aa").setFontWeight("bold"); sheet.setFrozenRows(1); } const id = data.id || Date.now().toString(); const row = [ id, data.date, data.time, data.pair, data.direction, data.entryPrice, data.stopLoss, data.takeProfit, data.lotSize, data.riskPercent, data.tradingType||"LIVE", data.setup, data.setupOther||"", data.timeframe, data.notes, data.isCompleted?"YES":"NO", data.result||"", data.closePrice||"", data.pnl||"", data.currency||"USD", data.closeDate||"", data.closeTime||"", data.duration||"", data.emotion||"", data.evaluation||"", data.accountId||"", data.accountName||"", data.broker||"", data.balanceAfter||"", new Date().toLocaleString("id-ID") ]; sheet.appendRow(row); sheet.autoResizeColumns(1, row.length); return jsonResponse({status:"success", id: id}); } function updateTrade(ss, data) { const sheet = ss.getSheetByName(SHEET_TRADES); if (!sheet) return jsonResponse({status:"error", message:"Sheet tidak ditemukan"}); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0].toString() === data.id.toString()) { const r = i + 1; const updates = { 15: data.isCompleted?"YES":"NO", 16: data.result||rows[i][15], 17: data.closePrice||rows[i][16], 18: data.pnl!==undefined ? data.pnl : rows[i][17], 20: data.closeDate||rows[i][19], 21: data.closeTime||rows[i][20], 22: data.duration||rows[i][21], 23: data.emotion||rows[i][22], 24: data.evaluation||rows[i][23], 27: data.balanceAfter!==undefined ? data.balanceAfter : rows[i][26] }; for (const [col, val] of Object.entries(updates)) { sheet.getRange(r, parseInt(col)).setValue(val); } return jsonResponse({status:"success", updated: data.id}); } } return jsonResponse({status:"error", message:"Trade tidak ditemukan"}); } function getTrades(ss, params) { const sheet = ss.getSheetByName(SHEET_TRADES); if (!sheet) return jsonResponse({status:"ok", trades:[]}); const data = sheet.getDataRange().getValues(); const headers = data[0]; const trades = data.slice(1).map(row => { const obj = {}; headers.forEach((h,i) => obj[h] = row[i]); return obj; }).filter(t => !params.accountId || t["Akun ID"] == params.accountId); return jsonResponse({status:"ok", trades}); } function saveAccount(ss, data) { let sheet = ss.getSheetByName(SHEET_ACCOUNTS); if (!sheet) { sheet = ss.insertSheet(SHEET_ACCOUNTS); const headers = ["ID","Nama Akun","Broker","No ID Akun","Server","Currency","Saldo Awal","Saldo Sekarang","Dibuat"]; sheet.getRange(1,1,1,headers.length).setValues([headers]) .setBackground("#1a1a2e").setFontColor("#00d4aa").setFontWeight("bold"); sheet.setFrozenRows(1); } const id = (data.id || Date.now()).toString(); // Cek apakah akun sudah ada (update, bukan duplikat) const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0].toString() === id) { sheet.getRange(i+1, 1, 1, 9).setValues([[ id, data.name, data.broker, data.accountId, data.server, data.currency, data.initialBalance, data.balance, rows[i][8] ]]); return jsonResponse({status:"success", id, updated: true}); } } // Akun baru sheet.appendRow([id, data.name, data.broker, data.accountId, data.server, data.currency, data.initialBalance, data.balance, new Date().toLocaleString("id-ID")]); return jsonResponse({status:"success", id, updated: false}); } function updateAccountBalance(ss, data) { const sheet = ss.getSheetByName(SHEET_ACCOUNTS); if (!sheet) return jsonResponse({status:"error", message:"Sheet tidak ditemukan"}); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0].toString() === data.accountId.toString()) { sheet.getRange(i+1, 8).setValue(data.newBalance); return jsonResponse({status:"success"}); } } return jsonResponse({status:"error", message:"Akun tidak ditemukan"}); } function getAccounts(ss) { const sheet = ss.getSheetByName(SHEET_ACCOUNTS); if (!sheet) return jsonResponse({status:"ok", accounts:[]}); const data = sheet.getDataRange().getValues(); const headers = data[0]; const accounts = data.slice(1).map(row => { const obj = {}; headers.forEach((h,i) => obj[h] = row[i]); return obj; }); return jsonResponse({status:"ok", accounts}); } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
4
Klik Deploy → New Deployment
5
Pilih type: Web App. Execute as: Me. Who has access: Anyone. Klik Deploy
6
Copy URL yang muncul (bentuknya panjang, mulai dengan https://script.google.com/macros/s/...)
7
Paste URL tersebut di tombol ⚙ Setup di pojok kanan atas, lalu klik Simpan.
🎉 Selesai! Setiap trade yang kamu simpan dari Web App ini akan otomatis masuk ke Google Sheets kamu dengan semua fitur baru.
